1. Mostrar todos los alumnos de la tabla alumnos

SELECT * FROM alumno

2. Mostrar los nombr dese los cursos con sus respectivas asignaturas

SELECT curso.nombreDescriptivo.asignatura.nombre
FROM curso INNER JOIN asignatura

3. Mostrar nombre y apellido de todos los alumnos que tengan beca

SELECT nombre, apellido1, apellido2, beca FROM alumno WHERE beca LIKE "si"

4. Obtener listado de profesores (nombre, apellido, numero de contacto)

SELECT profesor.nombre, profesor.apellido1, profesor.apellido2, tlfcontactoprof.telefono FROM profesor
INNER JOIN tlfcontactoprof ON profesor.idProfesor = tlfcontactoprof.idProfesor

5. Obtener listado de alumnos (todos los campos) que estan inscritos a la asignatura de programacion

SELECT al.* FROM alumno al
JOIN matricula m ON al.idAlumno = m.idAlumno JOIN asignatura a ON m.idAsignatura = a.idAsignatura WHERE a.nombre = 'Programacion'

6. Obtener listado de profesores y asignatura que imparten

SELECT p.idProfesor, p.nombre AS nombre_profesor, p.apellido1 AS apellido1_profesor, p.apellido2 AS apellido2_profesor, a.idAsignatura, a.nombre AS nombre_asignatura FROM profesor p
JOIN impartir i ON p.idProfesor = i.idProfesor JOIN asignatura a ON i.idAsignatura = a.idAsignatura ORDER BY p.idProfesor, a.idAsignatura;

7. Mostrar la asignatura que tiene mas alumnos inscritos y su total

SELECT a.idAsignatura, a.nombre, COUNT(m.idAlumno) AS total_alumnos FROM asignatura a JOIN matricula m ON a.idAsignatura = m.idAsignatura GROUP BYa.idAsignatura, a.nombre ORDER BY total_alumnos DESC LIMIT 1;

9. OBTENER EL LISTADO DE CURSOS, CON ASIGNATURA, Y EL MAESTRO QUE LAS IMPARTE

SELECT c.idCurso, c.nombreDescriptivo AS nombre_curso, a.idAsignatura, a.nombre AS nombre_asignatura, p.idProfesor, p.nombre AS nombre_profesor, p.apellido1 AS apellido1_profesor, p.apellido2 AS apellido2_profesor FROM curso c
JOIN asignatura a ON c.idCurso = a.curso JOIN impartir i ON a.idAsignatura = i.idAsignatura
JOIN profesor p ON i.idProfesor = p.idProfesor ORDER BY c.idCurso, a.idAsignatura, p.idProfesor
